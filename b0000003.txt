Attribute VB_Name = "modDescribeEDM"
Option Explicit

' =====================================================================
' DescribeEDM - reads the embedded images in a marketing email (EDM)
' and asks a locally/LAN-hosted vision-language model to describe them,
' for use when JAWS can't read image-only email content.
'
' SETUP
'   1. In Outlook: Alt+F11 -> File -> Import File -> select this .bas
'   2. Edit VLM_ENDPOINT and VLM_MODEL below to match your internal
'      OpenAI-compatible vision server.
'   3. Add DescribeCurrentEDM to the Quick Access Toolbar:
'      File > Options > Quick Access Toolbar > Choose commands from:
'      Macros > select DescribeCurrentEDM > Add.
'   4. Select or open the EDM email, then click the QAT button.
'
' No external references are required (Tools > References is untouched).
' All COM objects below are standard Windows 10 components accessed via
' late binding (CreateObject).
' =====================================================================

Private Const VLM_ENDPOINT As String = "http://REPLACE-WITH-INTERNAL-HOST:PORT/v1/chat/completions"
Private Const VLM_MODEL As String = "REPLACE-WITH-MODEL-NAME"
Private Const VLM_API_KEY As String = "doesnt-matter"
Private Const IGNORE_SSL_ERRORS As Boolean = True ' equivalent to verify_SSL=false / curl -k; only relevant for https:// endpoints
Private Const HTTP_TIMEOUT_MS As Long = 45000
Private Const VLM_PROMPT As String = "This image, or images are sequential fragments of one marketing email, in top-to-bottom, left-to-right reading order. Describe, in a natural paragraph, what the email is promoting, the main offer or call to action, and any important details such as prices or dates."

' Diagnostics: set False to stop writing the debug log (Debug.Print always runs).
Private Const DEBUG_LOG_ENABLED As Boolean = True

' Human-readable name of whatever stage is currently running, so ErrHandler
' can say *what* failed - VBA has no call stack to infer this from Err alone.
Private gCurrentStep As String

' ---------------------------------------------------------------------
' Entry point - assign this to a Quick Access Toolbar button.
'
' NOTE on debugging: for ErrHandler below to run at all (instead of the VBE
' breaking into the code), Tools > Options > General > Error Trapping must
' be set to "Break on Unhandled Errors", not "Break on All Errors".
' ---------------------------------------------------------------------
Public Sub DescribeCurrentEDM()
    Dim mailItem As Outlook.MailItem
    Dim images() As String
    Dim mimeTypes() As String
    Dim summary As String

    On Error GoTo ErrHandler
    DebugLog "===== DescribeCurrentEDM started ====="

RetryLabel:
    SetStep "Locating mail item"
    Set mailItem = GetActiveMailItem()
    If mailItem Is Nothing Then
        MsgBox "Please open or select an email first.", vbExclamation, "Describe EDM"
        Exit Sub
    End If

    SetStep "Extracting inline images"
    If Not ExtractInlineImages(mailItem, images, mimeTypes) Then
        MsgBox "No usable images were found in this email.", vbExclamation, "Describe EDM"
        Exit Sub
    End If
    DebugLog "Extracted " & (UBound(images) - LBound(images) + 1) & " image(s)"

    SetStep "Calling vision-language model"
    summary = CallVLM(images, mimeTypes)

    SetStep "Done"
    DebugLog "===== DescribeCurrentEDM succeeded ====="
    MsgBox summary, vbOKOnly + vbInformation, "Email Summary"
    Exit Sub

ErrHandler:
    ' Capture Err's fields into locals immediately - they get reset by the
    ' very next statement that can raise (even On Error Resume Next), and
    ' the handler itself must never be able to raise again (that's how you
    ' end up broken into the IDE on some unrelated line with no dialog).
    Dim errNum As Long, errDesc As String, errSrc As String, failedStep As String
    errNum = Err.Number
    errDesc = Err.Description
    errSrc = Err.Source
    failedStep = gCurrentStep
    On Error Resume Next

    DebugLog "FAILED at step '" & failedStep & "': Error " & errNum & " (" & errSrc & "): " & errDesc

    Dim msg As String
    msg = "Could not describe this email." & vbCrLf & vbCrLf & _
          "Step: " & failedStep & vbCrLf & _
          "Error " & errNum & " (" & errSrc & "):" & vbCrLf & errDesc
    If DEBUG_LOG_ENABLED Then
        msg = msg & vbCrLf & vbCrLf & "Details logged to:" & vbCrLf & LogFilePath()
    End If
    msg = msg & vbCrLf & vbCrLf & "Try again?"

    Dim retryResult As VbMsgBoxResult
    retryResult = MsgBox(msg, vbRetryCancel + vbCritical, "Describe EDM - Error")
    If retryResult = vbRetry Then
        ' Restore GoTo-ErrHandler mode before resuming - it was switched to
        ' Resume Next above so the handler itself can't break, but that mode
        ' would otherwise silently persist through the retried run too.
        On Error GoTo ErrHandler
        Resume RetryLabel
    End If
End Sub

' ---------------------------------------------------------------------
' Records the current stage (for ErrHandler) and appends it to the debug
' log, so a failure with no line-number info can still be traced to a step.
' ---------------------------------------------------------------------
Private Sub SetStep(ByVal stepName As String)
    gCurrentStep = stepName
    DebugLog "STEP: " & stepName
End Sub

Private Function LogFilePath() As String
    LogFilePath = Environ$("TEMP") & "\DescribeEDM_debug.log"
End Function

' ---------------------------------------------------------------------
' Appends a timestamped line to the debug log and the Immediate window.
' Must never itself raise - a logging failure should not mask the real
' error, so all failures here are swallowed.
' ---------------------------------------------------------------------
Private Sub DebugLog(ByVal text As String)
    Debug.Print text
    If Not DEBUG_LOG_ENABLED Then Exit Sub

    On Error Resume Next
    Dim f As Integer
    f = FreeFile
    Open LogFilePath() For Append As #f
    Print #f, Format$(Now, "yyyy-mm-dd hh:nn:ss") & "  " & text
    Close #f
End Sub

' ---------------------------------------------------------------------
' Locate the active/selected mail item, from an open Inspector or from
' the Explorer selection.
' ---------------------------------------------------------------------
Private Function GetActiveMailItem() As Outlook.MailItem
    Dim obj As Object

    On Error Resume Next
    If Not Application.ActiveInspector Is Nothing Then
        Set obj = Application.ActiveInspector.CurrentItem
    End If
    If obj Is Nothing Then
        If Not Application.ActiveExplorer Is Nothing Then
            If Application.ActiveExplorer.Selection.Count > 0 Then
                Set obj = Application.ActiveExplorer.Selection.Item(1)
            End If
        End If
    End If
    On Error GoTo 0

    If Not obj Is Nothing Then
        If obj.Class = olMail Then
            Set GetActiveMailItem = obj
        End If
    End If
End Function

' ---------------------------------------------------------------------
' Extract every inline (cid:) image referenced in the HTML body, in
' reading order. Falls back to any image-type attachment if no inline
' images are found.
' ---------------------------------------------------------------------
Private Function ExtractInlineImages(mailItem As Outlook.MailItem, ByRef images() As String, ByRef mimeTypes() As String) As Boolean
    Dim html As String
    Dim cids() As String
    Dim cidCount As Long
    Dim i As Long, searchPos As Long, cidPos As Long, endPos As Long
    Dim ch As String
    Dim cidValue As String

    html = mailItem.HTMLBody
    ReDim cids(0 To 50)
    cidCount = 0
    searchPos = 1

    Do
        cidPos = InStr(searchPos, html, "cid:", vbTextCompare)
        If cidPos = 0 Then Exit Do

        cidPos = cidPos + 4
        endPos = cidPos
        Do While endPos <= Len(html)
            ch = Mid(html, endPos, 1)
            If ch = """" Or ch = "'" Or ch = ">" Or ch = " " Then Exit Do
            endPos = endPos + 1
        Loop

        cidValue = Mid(html, cidPos, endPos - cidPos)
        If Len(cidValue) > 0 Then
            If Not IsInArray(cidValue, cids, cidCount) Then
                cids(cidCount) = cidValue
                cidCount = cidCount + 1
                If cidCount > UBound(cids) Then ReDim Preserve cids(0 To UBound(cids) + 50)
            End If
        End If

        searchPos = endPos + 1
    Loop

    If cidCount = 0 Then
        ExtractInlineImages = ExtractAttachmentImages(mailItem, images, mimeTypes)
        Exit Function
    End If

    Dim tempImages() As String
    Dim tempMimes() As String
    ReDim tempImages(0 To cidCount - 1)
    ReDim tempMimes(0 To cidCount - 1)

    Dim matched As Long
    matched = 0
    For i = 0 To cidCount - 1
        Dim att As Outlook.Attachment
        Dim b64 As String
        Dim mime As String
        Set att = Nothing
        SetStep "Encoding image " & (i + 1) & " of " & cidCount
        If FindAttachmentByCid(mailItem, cids(i), att) Then
            If EncodeAttachmentBase64(att, b64, mime) Then
                tempImages(matched) = b64
                tempMimes(matched) = mime
                matched = matched + 1
            End If
        End If
    Next i

    If matched = 0 Then
        ExtractInlineImages = ExtractAttachmentImages(mailItem, images, mimeTypes)
        Exit Function
    End If

    ReDim images(0 To matched - 1)
    ReDim mimeTypes(0 To matched - 1)
    For i = 0 To matched - 1
        images(i) = tempImages(i)
        mimeTypes(i) = tempMimes(i)
    Next i

    ExtractInlineImages = True
End Function

' ---------------------------------------------------------------------
' Fallback: pick up ordinary (non-inline) image attachments.
' ---------------------------------------------------------------------
Private Function ExtractAttachmentImages(mailItem As Outlook.MailItem, ByRef images() As String, ByRef mimeTypes() As String) As Boolean
    Dim att As Outlook.Attachment
    Dim count As Long
    Dim tempImages() As String
    Dim tempMimes() As String
    Dim ext As String
    Dim mime As String
    Dim b64 As String
    Dim i As Long

    If mailItem.Attachments.Count = 0 Then
        ExtractAttachmentImages = False
        Exit Function
    End If

    ReDim tempImages(0 To mailItem.Attachments.Count - 1)
    ReDim tempMimes(0 To mailItem.Attachments.Count - 1)
    count = 0

    For Each att In mailItem.Attachments
        If InStrRev(att.FileName, ".") > 0 Then
            ext = LCase(Mid(att.FileName, InStrRev(att.FileName, ".") + 1))
            mime = MimeTypeFromExtension(ext)
            If mime <> "" Then
                If EncodeAttachmentBase64(att, b64, mime) Then
                    tempImages(count) = b64
                    tempMimes(count) = mime
                    count = count + 1
                End If
            End If
        End If
    Next att

    If count = 0 Then
        ExtractAttachmentImages = False
        Exit Function
    End If

    ReDim images(0 To count - 1)
    ReDim mimeTypes(0 To count - 1)
    For i = 0 To count - 1
        images(i) = tempImages(i)
        mimeTypes(i) = tempMimes(i)
    Next i

    ExtractAttachmentImages = True
End Function

' ---------------------------------------------------------------------
' Match a cid value to an attachment via PR_ATTACH_CONTENT_ID.
' ---------------------------------------------------------------------
Private Function FindAttachmentByCid(mailItem As Outlook.MailItem, cidValue As String, ByRef att As Outlook.Attachment) As Boolean
    Const PR_ATTACH_CONTENT_ID As String = "http://schemas.microsoft.com/mapi/proptag/0x3712001E"
    Dim a As Outlook.Attachment
    Dim thisCid As String

    For Each a In mailItem.Attachments
        thisCid = ""
        On Error Resume Next
        thisCid = a.PropertyAccessor.GetProperty(PR_ATTACH_CONTENT_ID)
        On Error GoTo 0
        If StrComp(thisCid, cidValue, vbTextCompare) = 0 Then
            Set att = a
            FindAttachmentByCid = True
            Exit Function
        End If
    Next a

    FindAttachmentByCid = False
End Function

' ---------------------------------------------------------------------
' Save an attachment to a temp file, base64-encode it, then delete it.
' ---------------------------------------------------------------------
Private Function EncodeAttachmentBase64(att As Outlook.Attachment, ByRef b64 As String, ByRef mime As String) As Boolean
    Dim tempPath As String
    Dim ext As String

    If InStrRev(att.FileName, ".") = 0 Then
        EncodeAttachmentBase64 = False
        Exit Function
    End If

    ext = LCase(Mid(att.FileName, InStrRev(att.FileName, ".") + 1))
    mime = MimeTypeFromExtension(ext)
    If mime = "" Then
        EncodeAttachmentBase64 = False
        Exit Function
    End If

    tempPath = Environ$("TEMP") & "\edm_" & Format(Now, "yyyymmddhhnnss") & "_" & CStr(Int(Rnd * 100000)) & "." & ext
    att.SaveAsFile tempPath

    b64 = FileToBase64(tempPath)

    On Error Resume Next
    Kill tempPath
    On Error GoTo 0

    EncodeAttachmentBase64 = (Len(b64) > 0)
End Function

Private Function MimeTypeFromExtension(ext As String) As String
    Select Case LCase(ext)
        Case "jpg", "jpeg": MimeTypeFromExtension = "image/jpeg"
        Case "png": MimeTypeFromExtension = "image/png"
        Case "gif": MimeTypeFromExtension = "image/gif"
        Case "bmp": MimeTypeFromExtension = "image/bmp"
        Case Else: MimeTypeFromExtension = ""
    End Select
End Function

Private Function IsInArray(value As String, arr() As String, count As Long) As Boolean
    Dim i As Long
    For i = 0 To count - 1
        If StrComp(arr(i), value, vbTextCompare) = 0 Then
            IsInArray = True
            Exit Function
        End If
    Next i
    IsInArray = False
End Function

' ---------------------------------------------------------------------
' Base64-encode a file's bytes using the MSXML2.DOMDocument
' bin.base64 node trick (no third-party library needed).
' ---------------------------------------------------------------------
Private Function FileToBase64(filePath As String) As String
    Dim stream As Object
    Dim bytes() As Byte
    Dim domDoc As Object
    Dim node As Object

    Set stream = CreateObject("ADODB.Stream")
    stream.Type = 1 ' adTypeBinary
    stream.Open
    stream.LoadFromFile filePath
    bytes = stream.Read
    stream.Close

    Set domDoc = CreateObject("MSXML2.DOMDocument")
    Set node = domDoc.createElement("b64")
    node.DataType = "bin.base64"
    node.nodeTypedValue = bytes

    FileToBase64 = Replace(Replace(node.text, vbCr, ""), vbLf, "")
End Function

' ---------------------------------------------------------------------
' Call the internal OpenAI-compatible vision endpoint with all images
' in one request, and return the model's text response.
' ---------------------------------------------------------------------
Private Function CallVLM(images() As String, mimeTypes() As String) As String
    Dim http As Object
    Dim requestJson As String
    Dim responseText As String
    Dim content As String

    requestJson = BuildRequestJson(images, mimeTypes)
    DebugLog "Request: endpoint=" & VLM_ENDPOINT & " model=" & VLM_MODEL & _
             " images=" & (UBound(images) - LBound(images) + 1) & _
             " bytes=" & Len(requestJson)

    Set http = CreateObject("WinHttp.WinHttpRequest.5.1")
    http.Open "POST", VLM_ENDPOINT, False
    If IGNORE_SSL_ERRORS Then
        ' 13056 = SECURITY_FLAG_IGNORE_UNKNOWN_CA Or IGNORE_CERT_CN_INVALID Or
        '         IGNORE_CERT_DATE_INVALID Or IGNORE_CERT_WRONG_USAGE
        http.Option(4) = 13056 ' WinHttpRequestOption_SslErrorIgnoreFlags
    End If
    http.SetRequestHeader "Content-Type", "application/json"
    http.SetRequestHeader "Authorization", "Bearer " & VLM_API_KEY
    http.SetTimeouts HTTP_TIMEOUT_MS, HTTP_TIMEOUT_MS, HTTP_TIMEOUT_MS, HTTP_TIMEOUT_MS
    http.Send requestJson

    DebugLog "Response: status=" & http.Status & " " & http.StatusText

    If http.Status <> 200 Then
        DebugLog "Response body: " & http.ResponseText
        Err.Raise vbObjectError + 1, "CallVLM", "Server returned status " & http.Status & ": " & _
                  http.StatusText & vbCrLf & Left$(http.ResponseText, 800)
    End If

    responseText = http.ResponseText
    content = ExtractContentField(responseText)

    If Len(content) = 0 Then
        DebugLog "Response body (unparsed): " & Left$(responseText, 800)
        Err.Raise vbObjectError + 2, "CallVLM", "Could not parse a response from the server."
    End If

    DebugLog "Parsed content, length=" & Len(content)
    CallVLM = content
End Function

Private Function BuildRequestJson(images() As String, mimeTypes() As String) As String
    Dim sb As String
    Dim i As Long

    sb = "{"
    sb = sb & """model"":""" & JsonEscape(VLM_MODEL) & ""","
    sb = sb & """messages"":[{"
    sb = sb & """role"":""user"","
    sb = sb & """content"":["
    sb = sb & "{""type"":""text"",""text"":""" & JsonEscape(VLM_PROMPT) & """}"

    For i = LBound(images) To UBound(images)
        sb = sb & ",{""type"":""image_url"",""image_url"":{""url"":""data:" & mimeTypes(i) & ";base64," & images(i) & """}}"
    Next i

    sb = sb & "]"
    sb = sb & "}],"
    sb = sb & """max_tokens"":600,"
    sb = sb & """temperature"":0.2"
    sb = sb & "}"

    BuildRequestJson = sb
End Function

Private Function JsonEscape(s As String) As String
    Dim result As String
    result = s
    result = Replace(result, "\", "\\")
    result = Replace(result, """", "\""")
    result = Replace(result, vbCrLf, "\n")
    result = Replace(result, vbCr, "\n")
    result = Replace(result, vbLf, "\n")
    result = Replace(result, vbTab, "\t")
    JsonEscape = result
End Function

' ---------------------------------------------------------------------
' Purpose-built extractor for the first "content":"..." string in an
' OpenAI-compatible chat completion response, with JSON unescaping.
' ---------------------------------------------------------------------
Private Function ExtractContentField(json As String) As String
    Dim marker As String
    Dim startPos As Long, i As Long
    Dim ch As String
    Dim result As String
    Dim inEscape As Boolean

    marker = """content"":"""
    startPos = InStr(1, json, marker, vbTextCompare)
    If startPos = 0 Then Exit Function

    startPos = startPos + Len(marker)
    result = ""
    i = startPos
    inEscape = False

    Do While i <= Len(json)
        ch = Mid(json, i, 1)
        If inEscape Then
            Select Case ch
                Case "n": result = result & vbLf
                Case "t": result = result & vbTab
                Case "r": result = result & vbCr
                Case """": result = result & """"
                Case "\": result = result & "\"
                Case "/": result = result & "/"
                Case "u"
                    Dim hexCode As String
                    hexCode = Mid(json, i + 1, 4)
                    result = result & ChrW(CLng("&H" & hexCode))
                    i = i + 4
                Case Else
                    result = result & ch
            End Select
            inEscape = False
        Else
            If ch = "\" Then
                inEscape = True
            ElseIf ch = """" Then
                Exit Do
            Else
                result = result & ch
            End If
        End If
        i = i + 1
    Loop

    ExtractContentField = result
End Function
