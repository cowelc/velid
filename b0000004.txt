"""FastMCP server exposing GitLab wiki, issue, milestone, repository, and merge request operations as MCP tools.

Reads config from the environment (`.env` supported): `GITLAB_URL` and `GITLAB_PAT`
are required; `MCP_HOST`/`MCP_PORT` default to `0.0.0.0:8000`. On import it constructs
a shared `GitLabClient` and registers the tools, then serves them over SSE transport
when run as a script. Each `@mcp.tool()` function's docstring is the description shown
to MCP clients, so it is a thin async pass-through to the matching `GitLabClient` method.

Tool docstrings and parameter `Field(description=...)` annotations are written to be
*decision-oriented*: the first line says WHEN to reach for the tool, and read tools
state that they return live GitLab data (so a model answers by calling, not guessing).
The cross-project search tools (`search_wiki`, `search_issues`, `list_wiki_index`)
are backed by `search_cache.SearchCache` over the curated project list in
`GITLAB_WIKI_PROJECTS_FILE` (default `wiki_projects.toml`); the cache is warmed in
the background at startup and kept fresh by Events-API deltas.
"""
import argparse
import asyncio
import contextlib
import json
import os
from pathlib import Path
from typing import Annotated, Literal
from dotenv import load_dotenv
from pydantic import Field
from mcp.server.fastmcp import FastMCP
from client import GitLabClient, GitLabError
from search_cache import SearchCache, load_wiki_projects, log, split_sections, clip

load_dotenv()

_gitlab_url = os.environ.get("GITLAB_URL", "").rstrip("/")
_gitlab_pat = os.environ.get("GITLAB_PAT", "")
_host = os.environ.get("MCP_HOST", "0.0.0.0")
_port = int(os.environ.get("MCP_PORT", "8000"))
_projects_file = os.environ.get("GITLAB_WIKI_PROJECTS_FILE", "wiki_projects.toml")
# Snapshot of the search cache, restored on restart (set empty to disable persistence).
_cache_file = os.environ.get("GITLAB_SEARCH_CACHE_FILE", ".search_cache.json") or None


def _env_bool(name: str, default: bool = False) -> bool:
    """Read a boolean env var, treating 1/true/yes/on (any case) as True."""
    raw = os.environ.get(name)
    if raw is None:
        return default
    return raw.strip().lower() in ("1", "true", "yes", "on")


# Mutating tools (create_*/update_*/add_*_note) are OFF by default: the onboarding
# deployment is read-only, which also keeps the tool menu small for weaker models.
# Set GITLAB_MCP_ENABLE_WRITES=1 to expose them.
_enable_writes = _env_bool("GITLAB_MCP_ENABLE_WRITES", False)

if not _gitlab_url or not _gitlab_pat:
    raise RuntimeError("GITLAB_URL and GITLAB_PAT must be set (in .env or environment)")

gl = GitLabClient(_gitlab_url, _gitlab_pat)
cache = SearchCache(gl, load_wiki_projects(_projects_file), cache_file=_cache_file)
# Let every tool accept a human project *name*, not just an id/path: the client
# routes each `project` argument through the cache's resolver before calling GitLab.
gl.project_resolver = cache.resolve_project


# ── Reusable parameter annotations ────────────────────────────────────────────
# Shared so the same identifier is described the same way on every tool the model sees.

Project = Annotated[
    str,
    Field(
        description="Which project to act on: a numeric project ID (e.g. '42'), a "
        "'namespace/name' path (e.g. 'platform/backend'), OR a human project name "
        "(e.g. 'Integration of AXTRANS') — names are resolved automatically against "
        "the known projects. If a name is ambiguous the tool lists the matches; use "
        "find_project to look one up. Don't invent an id."
    ),
]
IssueIID = Annotated[
    int,
    Field(
        description="Project-local issue number — the '#123' shown in the GitLab UI and "
        "URLs (NOT the global issue id). Get it from list_issues/search_issues if unsure."
    ),
]
MRIID = Annotated[
    int,
    Field(
        description="Project-local merge request number — the '!45' shown in the GitLab UI "
        "and URLs (NOT the global mr id). Get it from list_merge_requests if unsure."
    ),
]


async def _warm_current_user():
    """Best-effort prime of the cached GET /user so 'me' resolves without a round-trip.

    Swallows failures (bad token, endpoint unreachable) so startup never depends on it.
    """
    try:
        await gl.get_current_user()
    except Exception as e:  # noqa: BLE001 - warm-up must never crash startup
        log(f"current-user warm-up skipped: {e}")


@contextlib.asynccontextmanager
async def _lifespan(_server):
    """Warm the search cache and the current-user identity in the background (non-blocking)."""
    warm = asyncio.create_task(cache.ensure_loaded()) if cache.projects else None
    asyncio.create_task(_warm_current_user())
    try:
        yield {}
    finally:
        if warm:
            warm.cancel()


mcp = FastMCP("gitlab", host=_host, port=_port, lifespan=_lifespan)


def write_tool():
    """Like `@mcp.tool()`, but only registers the tool when writes are enabled.

    The decorated function is always defined as a normal module attribute (so it can
    still be imported/tested), but it is only exposed over MCP when
    `GITLAB_MCP_ENABLE_WRITES` is set — otherwise the read-only deployment never sees it.
    """
    def decorator(fn):
        return mcp.tool()(fn) if _enable_writes else fn
    return decorator

@mcp.tool()
async def get_wiki_page(
    project: Project,
    slug: Annotated[
        str,
        Field(description="URL slug of the page (e.g. 'onboarding/setup') as returned by "
        "list_wiki_pages, search_wiki, or list_wiki_index."),
    ],
    section: Annotated[
        str | None,
        Field(description="Return only the section whose markdown heading matches this "
        "(case-insensitive, e.g. 'Rollback'); omit for the whole page. Use it to read one "
        "part of a long page without pulling the rest into context."),
    ] = None,
    max_chars: Annotated[
        int,
        Field(description="Cap on returned body length (default 8000); longer bodies are "
        "truncated with a marker so a large page can't overflow context.", ge=1),
    ] = 8000,
) -> dict:
    """Fetch the text of one specific wiki page (or one section of it).

    Call this whenever the user asks what a wiki/doc page says, or to read a page whose
    slug you got from list_wiki_pages/search_wiki. Returns the live page body — always
    call it instead of paraphrasing documentation from memory. For summarising across
    many pages, prefer gather_wiki (bounded extracts) over opening each page here."""
    page = await gl.get_wiki_page(project, slug)
    content = page.get("content") or ""
    if section:
        secs = split_sections(content, page.get("title") or "")
        wanted = section.strip().lower()
        picked = next(
            (s for s in secs if s["heading"].lower() == wanted
             or (s["path"] and s["path"][-1].lower() == wanted)),
            None,
        )
        if picked is None:
            return {
                "slug": slug,
                "error": f"no section matching {section!r}",
                "available_sections": [" > ".join(s["path"]) or s["heading"] for s in secs],
            }
        page["section"] = " > ".join(picked["path"]) or picked["heading"]
        content = picked["body"]
    page["content"] = clip(content, max_chars)
    return page

def _compact_todo(raw: dict) -> dict:
    """The subset of a to-do worth showing — what needs attention, where, and why."""
    target = raw.get("target") or {}
    project = raw.get("project") or {}
    body = raw.get("body") or ""
    return {
        "id": raw.get("id"),
        "action": raw.get("action_name"),      # assigned | mentioned | review_requested | ...
        "target_type": raw.get("target_type"),  # Issue | MergeRequest | ...
        "reference": target.get("references", {}).get("full") or target.get("iid"),
        "title": target.get("title"),
        "project": project.get("path_with_namespace") or project.get("name"),
        "author": (raw.get("author") or {}).get("name"),
        "state": raw.get("state"),
        "body": body if len(body) <= 200 else body[:200] + "…",
        "web_url": raw.get("target_url"),
        "updated_at": raw.get("updated_at"),
    }


@mcp.tool()
async def my_todos(
    state: Annotated[Literal["pending", "done"], Field(description="'pending' (default, still needs action) or 'done' (already dealt with).")] = "pending",
) -> list:
    """The current user's GitLab to-do list — everything awaiting THEIR attention across all projects.

    This is the tool for 'which MRs/issues/threads am I tagged or mentioned in', 'what needs
    my review', 'what's on my plate'. GitLab adds a to-do when you're @-mentioned, asked to
    review, or otherwise pulled in, so it answers those person-centric questions in one call —
    no project needed. It returns the WHOLE list; read the `action` and `target_type` on each
    item to see why it's there (mention, review request, …) and pick out what the user asked
    about — don't expect a pre-filtered result. Takes no required arguments — call with {} for
    the default pending list, or {"state": "done"} to see completed items.

    For 'issues ASSIGNED to me' specifically, prefer list_issues (omit its project so it
    defaults to assigned_to_me) — an issue you were assigned may not appear here (GitLab
    doesn't create a to-do for your own actions). Returns compact items (action, target_type,
    title, project, url); open the full thing with get_issue / get_merge_request using the
    project and reference shown."""
    todos = await gl.list_todos(state=state)
    return [_compact_todo(t) for t in todos]


# ── Issues ────────────────────────────────────────────────────────────────────

@mcp.tool()
async def list_issues(
    state: Annotated[Literal["opened", "closed", "all"], Field(description="Filter by state (default 'opened').")] = "opened",
    scope: Annotated[Literal["assigned_to_me", "created_by_me"] | None, Field(description="Global mode only (project omitted): whose issues to return; defaults to those assigned to you. Ignored when a project is given.")] = None,
    max_results: Annotated[int, Field(description="Global mode only: cap on issues returned (default 100). If more match, a trailing {'truncated': true, ...} marker is appended — narrow the query rather than assuming you saw everything.", ge=1)] = 100,
    assignee_username: Annotated[str | None, Field(description="Username of the assignee to filter by (no leading '@'). For 'assigned to me', prefer omitting project so scope defaults to assigned_to_me, or pass your own username from get_current_user.")] = None,
    milestone: Annotated[str | None, Field(description="Milestone title to filter by (exact title, not id).")] = None,
    labels: Annotated[str | None, Field(description="Comma-separated label names to require (e.g. 'bug,backend').")] = None,
    search: Annotated[str | None, Field(description="Free-text to match in issue title/description.")] = None,
    project: Annotated[str | None, Field(description="Project ID or 'namespace/name' to scope to ONE project. OMIT to search across ALL projects the token can see — in that global mode it returns issues assigned to you unless you set scope or assignee_username.")] = None,
) -> list:
    """List and filter issues, either within one project or across all projects.

    Call this whenever the user asks what issues exist, what's open or closed, what's
    assigned to someone, or which issues carry a label/milestone/keyword. For "issues
    assigned to me", OMIT project (scope defaults to assigned_to_me — no username needed);
    a self-referring assignee_username like 'me' is also accepted and resolved for you.
    Returns live issues from GitLab — always call rather than recalling from memory.
    Example arguments: {"state": "opened", "project": "platform/backend"}"""
    # Weak models often pass assignee_username="me" instead of omitting it; resolve any
    # self-reference to the real current username so the filter isn't looking for a user
    # literally named "me" (which would silently match nobody).
    if assignee_username and assignee_username.strip().lower() in {"me", "my", "mine", "myself", "self"}:
        assignee_username = (await gl.get_current_user())["username"]
    # Cap only the global path; a named project keeps the full (uncapped) listing.
    cap = None if project else max_results
    return await gl.list_issues(project, state, labels, milestone, assignee_username, search, scope=scope, max_results=cap)

# ── Activity ──────────────────────────────────────────────────────────────────

def _compact_event(raw: dict) -> dict:
    """One activity event trimmed to what/who/when (plus push detail when present)."""
    out = {
        "action": raw.get("action_name"),       # pushed to | opened | closed | commented on | ...
        "target_type": raw.get("target_type"),   # Issue | MergeRequest | WikiPage::Meta | Note | ...
        "title": raw.get("target_title"),
        "author": (raw.get("author") or {}).get("name"),
        "created_at": raw.get("created_at"),
    }
    push = raw.get("push_data")
    if push:
        out["push"] = {
            "ref": push.get("ref"),
            "commit_count": push.get("commit_count"),
            "commit_title": push.get("commit_title"),
        }
    note = raw.get("note")
    if note:
        body = note.get("body") or ""
        out["comment"] = body if len(body) <= 200 else body[:200] + "…"
    return out


@mcp.tool()
async def project_activity(
    project: Project,
    since: Annotated[str | None, Field(description="Only events on/after this date, 'YYYY-MM-DD' (day granularity). Omit for the most recent activity regardless of date.")] = None,
    limit: Annotated[int, Field(description="Max events to return, newest first (default 30).", ge=1, le=100)] = 30,
) -> list:
    """The recent activity feed for ONE project — pushes, issue/MR opens & closes, comments, wiki edits.

    This answers 'what's the latest activity / what's been happening / any recent updates on
    <project>'. You can name the project in plain language (e.g. 'Integration of AXTRANS') —
    it's resolved automatically. Returns compact events (action, target_type, title, author,
    time) newest first; drill into anything with get_issue / get_merge_request / get_commit."""
    events = await gl.list_project_events(project, after=since, per_page=min(limit, 100))
    return [_compact_event(e) for e in events[:limit]]

async def _searchable_cache() -> SearchCache:
    """Shared search-tool preamble: config check, first-run load, throttled delta."""
    if not cache.projects:
        raise RuntimeError(
            f"No search projects configured — create {_projects_file!r} with a "
            "`projects = [...]` list (see wiki_projects.toml) and restart."
        )
    await cache.ensure_loaded()
    cache.schedule_delta()  # async + throttled; this query is served from cache
    return cache

@mcp.tool()
async def search_code(
    search: Annotated[
        str | list[str],
        Field(description="A code token, or a list of tokens OR'd together. Must be literal "
        "identifiers/strings as they appear in source (e.g. ['AuthService','auth_service']), "
        "NOT a natural-language question — prose matches nothing."),
    ],
    projects: Annotated[list[str] | None, Field(description="Project paths/ids to limit the fan-out to (pass the few you've narrowed to); omit to sweep the whole curated set (slower).")] = None,
    extension: Annotated[str | None, Field(description="Restrict to a file extension, e.g. 'py' or 'ts' (no dot).")] = None,
    per_project: Annotated[int, Field(description="Max hits to pull from each project (default 5).", ge=1)] = 5,
    max_results: Annotated[int, Field(description="Max total hits to return across all projects (default 40).", ge=1)] = 40,
) -> list:
    """Find code across projects via GitLab Basic blobs search — a LITERAL `git grep`, NOT
    natural-language or semantic search.

    Call this when the user asks where a symbol/string/config lives in the code. Translate
    intent into real identifiers yourself (e.g. ['AuthService', 'auth_service', 'AuthClient']);
    a prose question like 'where do we call the auth service' will match nothing. Returns
    candidate hits (project, path, startline, ref, snippet) to judge yourself — not a ranked
    answer. Read a winner with get_file(project, path, ref). Searches the default branch HEAD
    only (no history, no other branches); binaries are skipped."""
    terms = [search] if isinstance(search, str) else list(search)
    if extension:
        terms = [f"{t} extension:{extension}" for t in terms]
    targets = projects if projects is not None else load_wiki_projects(_projects_file)
    if not targets:
        raise RuntimeError(
            f"No projects to search — pass `projects=[...]` or configure {_projects_file!r}."
        )

    sem = asyncio.Semaphore(12)

    async def one(proj, term):
        async with sem:
            try:
                return await gl.search_blobs(proj, term, per_page=per_project)
            except Exception as e:  # a dead/forbidden project must not sink the whole search
                # 403/404 = no access or no such project (expected, quiet); 5xx =
                # GitLab is actually failing, so surface it at a louder level.
                loud = isinstance(e, GitLabError) and e.server_error
                (log.error if loud else log.warning)("search_code skipping %r: %s", proj, e)
                return []

    batches = await asyncio.gather(*(one(p, t) for p in targets for t in terms))

    seen: set = set()
    out: list = []
    for hits in batches:
        for h in hits:
            key = (h["project_id"], h["path"], h["startline"])
            if key in seen:
                continue
            seen.add(key)
            out.append(
                {
                    "project": h["project_id"],
                    "path": h["path"],
                    "startline": h["startline"],
                    "ref": h["ref"],
                    "snippet": h["data"],
                }
            )
            if len(out) >= max_results:
                return out
    return out

# ── Operator cache admin (CLI, not MCP tools) ─────────────────────────────────
# The cache keeps itself fresh (background timer + a throttled delta on every
# search), so status/refresh are operator diagnostics, not something the model
# needs. They are exposed via `python server.py --cache-status` / `--refresh-cache`
# rather than as tools, to keep the model's tool menu focused on user questions.

async def search_cache_status() -> dict:
    """Health of the cross-project search cache: projects loaded/skipped, page/issue counts, sync recency."""
    return cache.status()


async def refresh_search_cache() -> dict:
    """Force an incremental sync now (wikis via Events API, issues via updated_after) and return the status."""
    if not cache.projects:
        raise RuntimeError(f"No search projects configured — create {_projects_file!r} first.")
    await cache.ensure_loaded()
    await cache.delta_sync()
    return cache.status()


# ── Entrypoint ────────────────────────────────────────────────────────────────

def main() -> None:
    """Run the MCP server, or perform a one-shot cache-admin command and exit."""
    parser = argparse.ArgumentParser(description="GitLab MCP server")
    group = parser.add_mutually_exclusive_group()
    group.add_argument(
        "--cache-status", action="store_true",
        help="Print search-cache health as JSON and exit (does not start the server).",
    )
    group.add_argument(
        "--refresh-cache", action="store_true",
        help="Force a cache sync, print the resulting health as JSON, and exit.",
    )
    args = parser.parse_args()

    if args.cache_status:
        async def _warm_then_status():
            if cache.projects:
                await cache.ensure_loaded()  # restore snapshot + catch up so counts are real
            return await search_cache_status()
        print(json.dumps(asyncio.run(_warm_then_status()), indent=2))
    elif args.refresh_cache:
        print(json.dumps(asyncio.run(refresh_search_cache()), indent=2))
    else:
        mcp.run(transport="sse")


if __name__ == "__main__":
    main()
